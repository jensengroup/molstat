    \frametitle{Gaussian Function}

\begin{lstlisting}
def gaussian(x):

    a = 1.0
    b = 0.0
    c = 1.0
    d = 0.0

    g = a*np.exp( -(x-b)**2 / (2*c**2) ) + d

    return g

\end{lstlisting}

\begin{lstlisting}
print gaussian(0.0)
\end{lstlisting}

\end{frame}
